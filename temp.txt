    container_name: whisper
    ports:
      - "4040:8000"
    volumes:
      # This acts as your cache. The model downloads here once and is reused on restarts.
      - ~/.cache/huggingface:/root/.cache/huggingface
    environment:
      - WHISPER__MODEL=Systran/faster-whisper-large-v3
    deploy:
      resources:
        reservations:
          devices:
            - driver: nvidia
              # Explicitly lock this container to GPU 0
              device_ids: ["0"]
              capabilities: [gpu]
    restart: unless-stopped
